R0 = 1./sigma;

%%Set up 4 resistor network for each point
for idx = 2:Nx-1
    for idy = 2:Ny-1
        R(idx,idy,1) = (R0(idx-1,idy)+R0(idx,idy))/2;
        R(idx,idy,2) = (R0(idx+1,idy)+R0(idx,idy))/2;
        R(idx,idy,3) = (R0(idx,idy+1)+R0(idx,idy))/2;
        R(idx,idy,4) = (R0(idx,idy-1)+R0(idx,idy))/2;
    end
end
